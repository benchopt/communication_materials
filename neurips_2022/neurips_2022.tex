\begin{frame}{Benchmarking algorithms is a pain}

Machine Learning relies on numerical validation

\vskip2em

Doing a benchmark:
\begin{itemize}
	\item takes time to reimplement methods
	\item takes time for reviewers to check reimplementations
	\item takes time to recode the tools needed
	\item all of this started from scratch by every submission
\end{itemize}
% \end{minipage}

\vskip2em


\Benchopt{} produces \textbf{reproducible}, \textbf{open}, \textbf{extendable} benchmarks

\end{frame}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%



%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{\Benchopt{}}

\Benchopt{} is an open source Python library to organize and run benchmarks
\begin{itemize}
    \item one base framework
    \item one repository per benchmark
\end{itemize}

\vskip1em
Already existing:
\vskip.5em
\begin{minipage}{0.45\linewidth}
    {\small
    \begin{itemize}
        \item Resnet18
        \item Lasso
        \item ICA
        \item Logistic regression
    \end{itemize}
    }
\end{minipage}
\begin{minipage}{0.48\linewidth}
    {\small
    \begin{itemize}
        \item Total Variation
        \item Ordinary Least Squares
        \item Non convex sparse regression
        \item linear SVM
    \end{itemize}
    }
\end{minipage}

\end{frame}

\begin{frame}{Structure of a benchmark}

    \begin{minipage}{0.45\linewidth}
        \dirtree{%
        .1 benchmark/.
        .2 objective.py.
        .2 datasets/.
        .3 dataset1.py.
        .3 dataset2.py.
        .2 solvers/.
        .3 solver1.py.
        .3 solver2.py.
    }
    \end{minipage}
    \begin{minipage}{0.45\linewidth}

    \textbf{Modular \& extendable}
    \vspace*{1mm}

    Adding a solver? add a file

    Adding a dataset? add a file
    \end{minipage}
